252async def fetch_with_retry(
253 source: DataSourceStrategy,
254 topic: str,
255 config: RetryConfig,
256 event_bus: EventBus
257) -> SourceResult:
258 """Wraps a data source fetch with exponential backoff."""
259
260 for attempt in range(config.max_retries + 1):
261 result = await source.fetch(topic)
262
263 if result.status == "success":
264 return result
265
266 if attempt < config.max_retries:
267 delay = min(
268 config.base_delay * (config.backoff_factor ** attempt)
269 + random.uniform(0, 0.5), # jitter
270 config.max_delay
271 )
272 await asyncio.sleep(delay)
