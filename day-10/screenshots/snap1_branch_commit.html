41class BranchValidator:
42 """Validates Git branch names according to standards."""
43 ALLOWED_PREFIXES = ["feature", "bugfix", "hotfix", "docs", "ci"]
44
45 @classmethod
46 def validate(cls, branch_name: str) -> bool:
47 parts = branch_name.split("/")
48 if len(parts) < 2 or not parts[0]:
49 raise BranchValidationError("Missing branch prefix")
50 prefix = parts[0]
51 if prefix not in cls.ALLOWED_PREFIXES:
52 raise BranchValidationError(f"Invalid prefix: {prefix}")
53 if branch_name != branch_name.lower():
54 raise BranchValidationError("Branch must be lowercase")
55 return True
